32 template <typename FunctionType = std::function<void()>>
33 requires std::invocable<FunctionType> &&

35 class thread_pool {
36 public:
37 thread_pool(const unsigned int &number_of_threads = std::thread::hardware_concurrency()) {
38 for (std::size_t i = 0; i < number_of_threads; ++i) {
39 threads_.emplace_back([&](const std::stop_token stop_tok) {
40 do {
41 // check if we have task
42 if (queue_.empty()) {
43 // no tasks, so we wait instead of spinning
44 std::unique_lock lock(condition_mutex_);
45 condition_.wait(lock, stop_tok, [this]() { return !queue_.empty(); });
46 }
47
48 // ensure we have a task before getting task
49 // since the dtor notifies via the condition variable as well
50 if (!queue_.empty()) {
51 // get the task
52 auto task = queue_.pop();
53 // invoke the task
54 std::invoke(std::move(task));
55 // decrement in-flight counter
56 --in_flight_;
57 }
58 } while (!stop_tok.stop_requested());
59 });
60 }
61 }

63 ~thread_pool() {
64 // wait for tasks to complete first
65 do {
66 std::this_thread::yield();
67 } while (in_flight_ > 0);
68
69 // stop all threads
70 for (auto &thread : threads_) {
71 thread.request_stop();
72 }

74 }

89 template <typename Function, typename... Args,
90 typename ReturnType = std::invoke_result_t<Function &&, Args &&...>>
91 requires std::invocable<Function, Args...>
92 [[nodiscard]] std::future<ReturnType> enqueue(Function f, Args... args) {
93 /*
94 * use shared promise here so that we don't break the promise later (until C++23)
95 *
96 * with C++23 we can do the following:
97 *
98 * std::promise<ReturnType> promise;
99 * auto future = promise.get_future();
100 * auto task = [func = std::move(f), ...largs = std::move(args),
101 promise = std::move(promise)]() mutable {...};
102 */
103 auto shared_promise = std::make_shared<std::promise<ReturnType>>();
104 auto task = [func = std::move(f), ... largs = std::move(args),
105 promise = shared_promise]() { promise->set_value(func(largs...)); };
106
107 // get the future before enqueuing the task
108 auto future = shared_promise->get_future();
109 // enqueue the task
110 enqueue_task(std::move(task));
111 return future;
112 }
